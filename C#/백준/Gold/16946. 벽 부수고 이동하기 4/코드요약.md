void ResultBFS(int startx, int starty, int tagnum)
{
    List<(int x, int y)> list = new List<(int, int)>();
    visited[startx, starty] = true;

    Queue<(int, int)> queue = new Queue<(int, int)>();
    queue.Enqueue((startx, starty));

    int count = 0;
    while (queue.Count > 0)
    {
        (int x, int y) = queue.Dequeue();
        list.Add((x, y));
        count++;
        for (int i = 0; i < 4; i++)
        {
            int px = x + dx[i];
            int py = y + dy[i];
            if (px < 0 || py < 0 || px >= m || py >= n || visited[px, py])
                continue;
            if (result[px, py] == 0)
            {
                queue.Enqueue((px, py));
            }
            visited[px, py] = true;
        }
    }
    foreach (var item in list)
    {
        result[item.x, item.y] = count;
        tag[item.x, item.y] = tagnum;
    }
}

int BFS(int startx, int starty)
{
    int count = 0;
    HashSet<int> taglist = new HashSet<int>();
    for (int i = 0; i < 4; i++)
    {
        int px = startx + dx[i];
        int py = starty + dy[i];
        if (px < 0 || py < 0 || px >= m || py >= n)
            continue;
        if (result[px, py] != -1)
        {
            int tagNumber = tag[px, py];
            if (taglist.Contains(tagNumber))
                continue;

            count += result[px, py];
            taglist.Add(tagNumber);
        }
    }
    return (count + 1) % 10;
}
